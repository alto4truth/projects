/* caps.c */

#include <stdio.h>
#include <ctype.h>

#define SEEK 0
#define REPLACE 1

int main(void)
{
int ch, state = SEEK;
while(( ch = getchar() ) != EOF )
{
switch( state )
{
case REPLACE:
switch( ch )
{
case ' ':
case '\t':
case '\n': state = SEEK;
break;
default: ch = tolower( ch );
break;
}
break;
case SEEK:
switch( ch )
{
case ' ':
case '\t':
case '\n': break;
default: ch = toupper( ch );
state = REPLACE;
break;
}
}
putchar( ch );
}
return 0;
}

The "puts()" function is like a simplified version of "printf()" without
format codes. It prints a string that is automatically terminated with a
newline:

puts( "Hello world!" );

The "fgets()" function is particularly useful: it reads a line of text
terminated by a newline. It is much less finicky about its inputs than "scanf()":
